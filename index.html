Integrasi Backend Terpadu

Google Apps Script (GAS) Setup Center

Gunakan kode backend ini untuk menghubungkan antarmuka ini langsung ke Google Spreadsheet Anda sebagai Database utama.

Panduan Langkah Deploy:

  1. Buatlah Google Spreadsheet baru di akun Google Drive Anda.
  2. Buat 8 sheet/tab dengan nama persis:
    Artikel, TentangKami, Informasi, Galeri, Pengaduan, Survey, Pegawai, Pengguna.
  3. Klik menu Ekstensi -> Apps Script di spreadsheet Anda.
  4. Hapus semua kode bawaan, lalu salin (copy) seluruh isi script di panel kanan dan paste ke dalam file Kode.gs.
  5. Klik Terapkan (Deploy) -> Penerapan Baru.
  6. Pilih Jenis penerapan Aplikasi Web. Atur akses ke "Siapa saja (Anyone)".
  7. Klik Deploy dan berikan izin otorisasi akses Google. Aplikasi siap digunakan secara online!

function doGet(e) {
  return HtmlService.createTemplateFromFile('index')
    .evaluate()
    .setTitle('DISDUKCAPIL Kabupaten Tolitoli')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1')
    .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL);
}

function setupDatabase() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  
  var sheets = {
    'Artikel': ['ID', 'Judul', 'Kategori', 'DeskripsiSingkat', 'Konten', 'GambarURL', 'VideoURL', 'Tanggal'],
    'TentangKami': ['Key', 'Nilai'],
    'Informasi': ['ID', 'Judul', 'Tipe', 'IsiAtauURL', 'InfoTambahan'],
    'Galeri': ['ID', 'Judul', 'GambarURL', 'Tanggal'],
    'Pengaduan': ['ID', 'Nama', 'NoHP', 'Email', 'IsiAduan', 'Status', 'Tanggal', 'CatatanAdmin'],
    'Survey': ['ID', 'Kepuasan', 'Komentar', 'Tanggal'],
    'Pegawai': ['ID', 'Nama', 'NIP', 'Jabatan', 'Bidang', 'PhotoURL'],
    'Pengguna': ['ID', 'Username', 'Password', 'NamaLengkap', 'Role']
  };
  
  for (var name in sheets) {
    var sheet = ss.getSheetByName(name);
    if (!sheet) {
      sheet = ss.insertSheet(name);
      sheet.appendRow(sheets[name]);
    }
  }
}

// Global CRUD Helpers untuk dipanggil oleh AJAX Frontend
function readData(sheetName) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) return [];
  
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var rows = data.slice(1);
  
  return rows.map(function(row) {
    var obj = {};
    headers.forEach(function(header, idx) {
      obj[header] = row[idx];
    });
    return obj;
  });
}

function addData(sheetName, obj) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) return false;
  
  var headers = sheet.getDataRange().getValues()[0];
  var newRow = headers.map(function(header) {
    return obj[header] !== undefined ? obj[header] : '';
  });
  
  sheet.appendRow(newRow);
  
  if (sheetName === 'Pengaduan') {
    handleAduanNotification(obj);
  }
  return true;
}

function updateData(sheetName, id, obj) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) return false;
  
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var idIdx = headers.indexOf('ID');
  if (idIdx === -1) idIdx = headers.indexOf('Key');
  
  for (var i = 1; i < data.length; i++) {
    if (data[i][idIdx].toString() === id.toString()) {
      headers.forEach(function(header, hIdx) {
        if (obj[header] !== undefined) {
          sheet.getRange(i + 1, hIdx + 1).setValue(obj[header]);
        }
      });
      return true;
    }
  }
  return false;
}

function deleteData(sheetName, id) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(sheetName);
  if (!sheet) return false;
  
  var data = sheet.getDataRange().getValues();
  var headers = data[0];
  var idIdx = headers.indexOf('ID');
  
  for (var i = 1; i < data.length; i++) {
    if (data[i][idIdx].toString() === id.toString()) {
      sheet.deleteRow(i + 1);
      return true;
    }
  }
  return false;
}

function handleAduanNotification(aduan) {
  var adminEmail = "disdukcapil@tolitolikab.go.id";
  var emailSubject = "[ADUAN WARGA BARU] " + aduan.ID;
  var emailBody = "Ada laporan masuk baru dari warga:\n\n" +
                  "No. Tiket: " + aduan.ID + "\n" +
                  "Nama Pelapor: " + aduan.Nama + "\n" +
                  "WhatsApp: " + aduan.NoHP + "\n" +
                  "Email: " + aduan.Email + "\n" +
                  "Isi Laporan: " + aduan.IsiAduan + "\n\n" +
                  "Harap segera diproses melalui portal manajemen SIP-DUKCAPIL.";
                  
  try {
    MailApp.sendEmail(adminEmail, emailSubject, emailBody);
  } catch (e) {
    Logger.log("Gagal mengirim email: " + e.toString());
  }
}
